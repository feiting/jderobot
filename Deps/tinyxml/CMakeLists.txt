
FIND_PATH( LIBTINYXML_INCLUDE_DIR NAMES tinyxml.h PATHS ENV C++LIB ENV PATH PATH_SUFFIXES include/xml/)

IF( LIBTINYXML_INCLUDE_DIR )

    MESSAGE("*** LIBTINYXML ok")
    FIND_LIBRARY( LIBTINYXML_LIBRARY1 NAMES tinyxml PATHS ENV C++LIB ENV PATH PATH_SUFFIXES lib lib64)
   
    SET (LIBTINYXML_LIBRARIES ${LIBTINYXML_LIBRARY1} )
    IF( LIBTINYXML_LIBRARIES )
		#MESSAGE("LIBTINYXML LIBRARY FOUND IN ${LIBTINYXML_LIBRARIES}")
	SET(DEPS "${DEPS} libtinyxml2.6.2,")
    ENDIF(LIBTINYXML_LIBRARIES )

ENDIF(LIBTINYXML_INCLUDE_DIR)

IF (NOT  LIBTINYXML_INCLUDE_DIR)
	MESSAGE("*** LIBTINYXML not found")
ELSE()
	link_directories(${LIBTINYXML_LIBRARIES})
	include_directories(${LIBTINYXML_INCLUDE_DIR})
ENDIF()
